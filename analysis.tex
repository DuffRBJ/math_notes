\section{Helpful Lemmas \& Inequalities}
\begin{lem}
\begin{itemize}
    \item Let $ x,y\in \R $. If for all $ \varepsilon>0 $ we have $ x-y<\varepsilon $, then $x\leq y$.
    \item Let $x,y \in \R$. If for all $ \varepsilon>0 $  we have $ |x-y|<\varepsilon $, then $ x=y $.
\end{itemize}
\end{lem}

Proof: Note that if $ x>y $ then $ x-y\geq \frac{x-y}{2}=\varepsilon $, which proves the contrapositve.
The second part followsw from the first since $ |x-y|<\varepsilon \implies x-y<\varepsilon \land y-x <\varepsilon$.


\section{Differentiation}

\begin{defn}[Compactness]
    A set $S\subset U$ (in a topological space) is called \textbf{compact} if for every open cover $ \mathcal{T} $ of $ S $ there is a finite subcover of $ S $. 
\end{defn}


\begin{thm}\label{thm-continuous_compact}
The image of a compact set under a continuous function is compact.
\end{thm}
Proof idea: If $ f:D\to \R $ is a continuous function and $ D $ is compact, then let $ \mathcal{T} $ be any open cover of $ f(D) $. Then taking the preimage of each $ T\in \mathcal{T} $ gives an open cover of $D$. This has a finite subcover, $ \mathcal{T}_0 $. Then mapping these back to elements of $ \mathcal{T} $ gives a finite subcover of $ f(D) $.

\begin{thm}
Let $ S $ be a compact set on $ \R $ and $ f:S\to \R $ be continuous. Then $ f $ achieves both its maximum and minimum on $ S $. 
\end{thm}

Proof: By \ref{thm-continuous_compact} theorem, $ f(S) $ is compact. By Heine-Borel, it is closed and bounded and hence $ \sup f(S)=\max f(S)$ and likewise for the minimum. $ \square $  

\begin{thm}[Rolle's Theorem]
    Let $ f:\R \to \R $ be a function continuous on $ [a,b] $ and differentiable on $ (a,b) $ such that $ f(a)=f(b) $. Then there exists $ c \in (a,b) $ s.t. $ f'(c)=0 $. 
\end{thm}

\begin{thm}[Mean Value Theorem]
    Let $f:\R \to \R$ be a function that is continuous on $[a,b]$ and differentiable on $(a,b)$. Then there exists a $c\in (a,b)$ such that
   $$ f'(c)= \frac{f(b)-f(a)}{b-a} $$  
\end{thm}

Proof: Consider the function $g(x)=f(x)- \frac{f(b)-f(a)}{b-a}(x-a)$. Then $ g(a)=f(a) $ and $ g(b)=f(a) $. Thus by Rolle's Theorem there exists a $c\in (a,b)$ s.t. $ g'(c)=0 $ and the rest follows. $\square$

\begin{thm}[Taylor's Theorem]
    Let $ f: \R \to \R $ be a  function that has its first $n$ derivatives continuous on $[a,b]$ and differentiable on $ (a,b) $. Fix a point to expand around, $ x_0 $. Let $ x \in [a,b]$ with $ x\neq x_0 $. Then for any $n\in \N$ there exists and $c \in (x,x_0)$ such that
    $$f(x)= \sum_{k=0}^{n} \frac{f^{(n)}(x_0)}{k!} (x-x_0)^k + \frac{f^{(n+1)}(c)}{(n+1)!}(x-x_0)^{n+1}$$  

\end{thm}

Idea of Proof:
\begin{itemize}
    \item Construct a function $F(t)$ such that 
    \begin{enumerate}
	\item $ F $ is continuous on $ (a,b) $ and differentiable on $ [a,b] $.
	\item $F(x)=f(x)$ and $ F(x_0)=f(x) $ 
	\item $F'(t)=0$ forces the correct coefficient of $ (x-x_0)^{n+1} $ 
    \end{enumerate}
\item Use the mean value theorem on $F(t)$.
\end{itemize}


Noteworthy property of Modified Taylor Series:
If $P(t)= \sum_{k=0}^{n} \frac{f^{(k)}(t)}{k!} (x-t)^k$ 
then $$ P'(t)= \frac{f^{(n+1)}(t)}{n!}(x-t)^{n} $$ 


Proof:( following Lay 28.2)
Let $ x\in [a,b]$ and fix $ n\in \N $. Let
$$F(t)= \sum_{k=0}^{n} \frac{f^{(k)}(t)}{k!}(x-t)^k + M(x-t)^{n+1}$$ 
where $M$ is the unique solution to 
$$ f(x)= \sum_{k=0}^{n} \frac{f^{(k)}(x_0)}{k!} (x-x_0)^k +M(x-x_0)^{n+1}.  $$ 
Note that by the noteworthy property mentioned above we have
$$ F'(t)= \frac{f^{(n+1)}(t)}{n!}(x-t)^n + (n+1)M(x-t)^n.$$ 
Note also that $F(x)=F(x_0)=f(x)$, so invoking the mean value theorem (or Rolle's Theorem) we obtain that there exists a $ c $ between $x$ and $ x_0 $ strictly such that
$$ F'(t)=0=\frac{f^{(n+1)}(t)}{n!}(x-t)^n + (n+1)M(x-t)^n.$$ 
Solving for $M$ gives $M= \frac{f^{(n+1)}(x_0)}{(n+1)!}$ as required. $\square$

\begin{ex}
    Let $f(x)=x^2+2x+1$ on $ [0,3] $ with $x_0=0$. Let $n=1$. Then there should exist a $c$ such that $$ f(2)=1+2(x-0)+ \frac{f^{(n+1)}(c)}{(n+1)!}(x-0)^{n+1}
$$ 
$$ 1+2x+ \frac{2}{2}x^2$$ 
so we recover our polynomial.
\end{ex}

We now note that we can rephrase this theorem in terms of a remainder.

\begin{thm}[Taylor's Remainder Theorem]
    Let $f:\R\to \R$ be $n+1$ times differentiable and $x_0\in \R$. Then if we approximate
$$ f(x)\approx p_n(x) = \sum_{k=0}^{n} \frac{f^{(k)}(x_0)}{k!}(x-x_0)^n  $$ 
    then their exists a $c$ between $ x $ and $ x_0 $ such that
   $$ f(x)-p_n(x)= \frac{f^{(n+1)}(c)}{(n+1)!}(x-x_0)^{n+1}.$$ 
   Note thought, that $ c $ depends on $ x $.
\end{thm}

\begin{rem}
Note that if $f$ is a polynomial of degree $k$, then $p_n=f$ for all $n\geq k$ and all $x_0\in \R$.
\end{rem}

\begin{lem}
    Let $f$ exist in a neighborhood of $c$ and that $f''(c)$ exists. Then $$ f''(c)= \lim \limits_{h\to 0} \frac{f(c+h)+f(c-h)-2f(c)}{h^2} $$ 
    $$  f''(c)= \lim \limits_{h\to 0} \frac{(f(c+h)-f(c))+(f(c-h)-f(c))}{h^2}$$ 
\end{lem}

Proof: Starting with the limit and using L'Hospital's rule we get:

    $$  \lim \limits_{h\to 0} \frac{f(c+h)+f(c-h)-2f(c)}{h^2}=  \lim \limits_{h\to 0} \frac{f'(c+h)-f'(c-h)}{2h}$$ 
    $$ =\lim_{h \to 0} \frac{f'(c+h)-f'(c)+ (f'(c)-f'(c-h))}{2h} $$ 
    $$ = \frac{2f''(c)}{2} = f''(c)  \square.$$ 

    \begin{lem} The following are equivalent definitions of the derivative.
	\begin{itemize}
\item The usual one
\item $ f'(c)= \lim_{x\to c} \frac{f(x)-f(c)}{x-c} $  
	    \item   $ f'(c)=\lim_{h\to 0} \frac{f(c+h)-f(c-h)}{2h} $ 
	\end{itemize}
  
    \end{lem}

    \section{Integration}
    \begin{defn}
	A partition of $ [a,b] $ is an ordered, finite subset $ P $ of $ [a,b] $ indexed from $ 0  $ to $ n $. 
	$$ P = \{ a=x_0, x_1, x_2, \ldots, x_n=b \} $$ 
    \end{defn}

    \begin{defn}
	For a partition $ P $ of $ [a,b] $ define the \textbf{Upper Sum} of a bounded functino $f$ to be
	$$ U(f,P,[a,b])= \sum_{i=1}^{n} \Delta x_i \cdot \sup \{ f(x): x\in [x_{i-1}, x_i] \}  $$ 
	Similarly, the lower sum is
	
	$$ L(f,P,[a,b])= \sum_{i=1}^{n} \Delta x_i \cdot \inf \{ f(x): x\in [x_{i-1}, x_i] \}  $$ 
    \end{defn}

    \begin{defn}
    The Riemman Integral of $ f $ if it exists is 
    $$ \int_{a}^{b} f(x)  dx =\inf_{Q\geq P} U(f,Q,[a,b]) = \sup_{Q\geq P} L(f,Q,[a,b]) $$ 
   for any starting $ P $ (or just $ \{ a,b \} $  if the two numbers agree. 
    \end{defn}
    
    It may be interesting to note that  the set of partitions is a sublattice of the lattice of sets with $ \cup $ as a join and $ \cap $  as a meet (or vice versa).

    \begin{lem}
    Note that for any two partitions $P$, $Q$ we have
    $$ L(f,P) \leq U(f,Q) $$ 
    \end{lem}

    Proof: Consider the join $ T=P\cup Q $, then by monotonicity of $ L $  and $ U $ under refinement we have 
    $$ L(F,P) \leq L(f,T)\leq U(f,T)\leq U(f,Q) $$ . $ \square $ 

    \begin{ex}
	Note the Dirichlet function is not integrable by showing $ U(f)=1 $ but $ L(f)=0 $ over $ [0,1] $.
    \end{ex}
